@st.cache_data
def load_and_score_data():
    """
    Load data and calculate a high-contrast Health Index using 
    Percentile Ranking to ensure a wide 0-100 spread.
    """
    df = pd.read_csv('Global Health Statistics.csv').dropna()
    
    # Step 1: Calculate average metrics per country first
    # This prevents row-level noise from diluting the differences
    country_stats = df.groupby('Country').agg({
        'Healthcare Access (%)': 'mean',
        'Mortality Rate (%)': 'mean',
        'Doctors per 1000': 'mean',
        'Per Capita Income (USD)': 'mean'
    })

    # Step 2: Scale components using Percentile Ranks (0-100)
    # This forces the data to use the full range regardless of outliers
    def get_rank_score(series, invert=False):
        ranks = series.rank(pct=True) * 100
        return 100 - ranks if invert else ranks

    access_r = get_rank_score(country_stats['Healthcare Access (%)'])
    mortality_r = get_rank_score(country_stats['Mortality Rate (%)'], invert=True)
    docs_r = get_rank_score(country_stats['Doctors per 1000'])
    income_r = get_rank_score(np.log1p(country_stats['Per Capita Income (USD)']))

    # Step 3: Weighted Combined Rank
    raw_weighted = (
        (access_r * 0.4) + 
        (mortality_r * 0.3) +
        (docs_r * 0.2) + 
        (income_r * 0.1)
    )

    # Step 4: Final Percentile Stretch
    # This guarantees the top country is 100 and the bottom is 0
    final_scores = raw_weighted.rank(pct=True) * 100
    
    # Map these distinct scores back to the main dataframe
    score_dict = final_scores.to_dict()
    df['Health_Index'] = df['Country'].map(score_dict)
        
    return df

def load_model():
    if os.path.exists('health_model.pkl'):
        return joblib.load('health_model.pkl')
    return None

def get_status(score):
    if score >= 75: return "Safe / Excellent", "#2ecc71", "✅"
    if score >= 40: return "Moderate / Stable", "#f1c40f", "⚠️" # Adjusted threshold for better distribution
    return "High Risk / Critical", "#e74c3c", "🚨"
